enum HTML::node::node_type

Tree node types.

66 {
67 node_null, // Empty (null) node handle
68 node_document, // A document tree's absolute root
69 node_element, // Element tag, i.e. '<node/>'
70 node_attribute, // A tree node's attribute
71 node_pcdata, // Plain character data, i.e. 'text'
72 node_cdata, // Character data, i.e. '<![CDATA[text]]>'
73 node_comment, // Comment tag, i.e. '<!-- text -->'
74 node_pi, // Processing instruction, i.e. '<?name?>'
75 node_declaration, // Document declaration, i.e. '<?html version="1.0"?>'
76 node_doctype // Document type declaration, i.e. '<!DOCTYPE doc>'
77 };
