293 /**
294 * Find node from subtree using predicate. Returns first node from
295 * subtree (depth-first), for which predicate returned true.
296 *
297 * @cond INTERNAL
298 */
299 template <typename T> Node find_node(T pred) const {
300 Node child = this->first_child();
301 int traverse_depth = 1;
302
303 while(traverse_depth > 0) {
304 if (pred(child))
305 return child;
306 if (child->first_child()) {
307 child = child->first_child();
308 ++traverse_depth;
309 }
310 else if (child->next_sibling())
311 child = child->next_sibling();
312 else {
313 while(!child->next_sibling() && traverse_depth > 0) {
314 child = child->parent();
315 --traverse_depth;
316 }
317 if (traverse_depth > 0)
318 child = child->next_sibling();
319 }
320 }
321
322 return Node(nullptr);
323 }
324 /* @endcond */

591/**
592 * HTML parser.
593 *
594 * For testing purposes some functions are exposed publicly. Like DOCTYPE parsing.
595 * DOCTYPE consists of nested sections of the following possible types:
596 * 1. <!-- ... -->, <? ... ?>, "...", '...'
597 * 2. <![...]]>
598 * 3. <!...>
599 * First group can not contain nested groups.
600 * Second group can contain nested groups of the same type.
601 * Third group can contain all other groups.
602 *
603 * NOTE:
604 * Actualy HTML standard only includes the third group <!DOCTYPE ...>. So the
605 * first two groups parsing might be removed in the future.
606 */
607class parser {
608public:
609 /**
610 * chartype_t is shared between parser and tokenizer.
611 */
612 enum chartype_t {
613 ct_parse_pcdata = 1, // Parse until \0, <
614 ct_parse_attr = 2, // Parse until \0, \r, \t, \n, >, space
615 ct_parse_attr_ws = 4, // Parse until \0, &, \r, ', ", \n, tab
616 ct_space = 8, // Parse until \r, \n, space, tab
617 ct_parse_cdata = 16, // Parse until \0, ], >, \r
618 ct_parse_comment = 32, // Parse until \0, -, >, \r
619 ct_symbol = 64, // Any symbol > 127, a-z, A-Z, 0-9, _, :, -, .
620 ct_start_symbol = 128 // Any symbol > 127, a-z, A-Z, _, :
621 };
622
623 /**
624 * Parsing status, shared betweeen parser and parse_error.
625 */
626 enum parse_status {
627 status_ok = 0, // No error
628 status_file_not_found, // File was not found during load_file()
629 status_io_error, // Error reading from file/stream
630 status_out_of_memory, // Could not allocate memory
631 status_internal_error, // Internal error occurred
632 status_unrecognized_tag, // Parser could not determine tag type
633 status_bad_pi, // Parsing error occurred while parsing document declaration/processing instruction
634 status_bad_comment, // Parsing error occurred while parsing comment
635 status_bad_cdata, // Parsing error occurred while parsing CDATA section
636 status_bad_doctype, // Parsing error occurred while parsing document type declaration
637 status_bad_pcdata, // Parsing error occurred while parsing PCDATA section
638 status_bad_start_element, // Parsing error occurred while parsing start element tag
639 status_bad_attribute, // Parsing error occurred while parsing element attribute
640 status_bad_end_element, // Parsing error occurred while parsing end element tag
641 status_end_element_mismatch // There was a mismatch of start-end tags (closing tag had incorrect name, some tag was not closed or there was an excessive closing tag)
642 };
643
644 // Parsing options
645
646 /**
647 * Minimal parsing mode (equivalent to turning all other flags off).
648 * Only elements and PCDATA sections are added to the DOM tree, no text
649 * conversions are performed.
650 */
651 static const unsigned int parse_minimal = 0x0000;
652
653 /**
654 * This flag determines if processing instructions (node_pi) are added
655 * to the DOM tree. This flag is off by default.
656 */
657 static const unsigned int parse_pi = 0x0001;
658
659 /**
660 * This flag determines if comments (node_comment) are added to the
661 * DOM tree. This flag is off by default.
662 */
663 static const unsigned int parse_comments = 0x0002;
664
665 /**
666 * This flag determines if CDATA sections (node_cdata) are added to
667 * the DOM tree. This flag is on by default.
668 */
669 static const unsigned int parse_cdata = 0x0004;
670
671 /**
672 * This flag determines if plain character data (node_pcdata) that
673 * consist only of whitespace are added to the DOM tree.
674 * This flag is off by default; turning it on usually results in slower
675 * parsing and more memory consumption.
676 * NOTE: currently this option is disabled.
677 */
678 static const unsigned int parse_ws_pcdata = 0x0008;
679
680 /**
681 * This flag determines if character and entity references are expanded
682 * during parsing. This flag is on by default.
683 */
684 static const unsigned int parse_escapes = 0x0010;
685
686 /**
687 * This flag determines if EOL characters are normalized
688 * (converted to 0xA) during parsing. This flag is on by default.
689 */
690 static const unsigned int parse_eol = 0x0020;
691
692 /**
693 * This flag determines if attribute values are normalized using CDATA
694 * normalization rules during parsing. This flag is on by default.
695 */
696 static const unsigned int parse_wconv_attribute = 0x0040;
697
698 /**
699 * This flag determines if attribute values are normalized using
700 * NMTOKENS normalization rules during parsing. This flag is off by default.
701 */
702 static const unsigned int parse_wnorm_attribute = 0x0080;
703
704 /**
705 * This flag determines if document declaration (node_declaration) is
706 * added to the DOM tree. This flag is off by default.
707 */
708 static const unsigned int parse_declaration = 0x0100;
709
710 /**
711 * This flag determines if document type declaration (node_doctype) is
712 * added to the DOM tree. This flag is off by default.
713 */
714 static const unsigned int parse_doctype = 0x0200;
715
716 /**
717 * The default parsing mode.
718 * Elements, PCDATA and CDATA sections are added to the DOM tree,
719 * character/reference entities are expanded, End-of-Line characters
720 * are normalized, attribute values are normalized using CDATA
721 * normalization rules.
722 */
723 static const unsigned int parse_default = parse_cdata
724 | parse_escapes
725 | parse_wconv_attribute
726 | parse_eol;
727
728 /**
729 * The full parsing mode.
730 * Nodes of all types are added to the DOM tree, character/reference
731 * entities are expanded, End-of-Line characters are normalized,
732 * attribute values are normalized using CDATA normalization rules.
733 */
734 static const unsigned int parse_full = parse_default
735 | parse_pi
736 | parse_comments
737 | parse_declaration
738 | parse_doctype;
739
740public:
741
742 /**
743 * Creates new parser with the specified parsing options.
744 */
745 parser(unsigned int options = parse_default);
746
747 /**
748 * Parses the specified HTML string and returns document object
749 * representing the HTML document tree.
750 * \note
751 * Capitalizes element and attribute names.
752 */
753 Document parse(const String& str_html);
754
755 /**
756 * @return last parse status description.
757 */
758 String status_description() const;
759
760 /**
761 * @return parse status string representation.
762 */
763 static String status_description(parse_status status);
764
765 /**
766 * Returns the parsed HTML document object.
767 */
768 Document get_document() const;
769
770private:
771 unsigned int options_;
772 parse_status status_ = status_ok;
773 Char* error_offset_ = nullptr;
774 Document document_;
775 Node current_node_;
776
777 /**
778 * Checks if the specified parsing option is set.
779 */
780 bool option_set(unsigned int opt);
781
782 /**
783 * Parse node contents, starting with exclamation mark.
784 *
785 * @param opts parse options. E.g. parse comments or not, etc.
786 * @param endch character by which the doctype element must end.
787 * Otherwise the parse_error is thrown.
788 */
789 const Char* parse_exclamation(const Char* s, Char endch = '>');
790
791 /**
792 * Parses the first group of doctype primitives:
793 * 1. <!-- ... -->, <? ... ?>, "...", '...'
794 * This method simply advances the specified string pointer to the next
795 * symbol of the end of doctype element.
796 *
797 * @throws parse_error on failure.
798 */
799 static const Char* advance_doctype_primitive(const Char* s);
800
801 /**
802 * Parses the second group of doctype:
803 * 2. <![...]]>
804 * This method simply advances the specified string pointer to the
805 * character which is post the last char of doctype element.
806 *
807 * @throws parse_error on failure.
808 */
809 static const Char* advance_doctype_ignore(const Char* s);
810
811 /**
812 * Parses the third group of doctype:
813 * 3. <!...>
814 * This method simply advances the specified string pointer to the
815 * character which is post the last char of doctype element.
816 *
817 * @param endch character by which the doctype element must end.
818 * Otherwise the parse_error is thrown.
819 * @param top_level flag indicating if this method was called
820 * from top level or recursively from itself.
821 */
822 static const Char* advance_doctype_group(const Char* s,
823 Char endch,
824 bool top_level = true);
825};
